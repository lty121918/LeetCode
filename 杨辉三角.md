# 杨辉三角

两个for循环打印，在for循环中规定最开始的两行和每一行的首位和末位是1，循环上一行的两个位置加起来，作为下一行的值。

```js
/**
 * @param {number} numRows
 * @return {number[][]}
 */
var generate = function(numRows) {
    let result = [];
    for (let i = 0; i < numRows; i++) {
        if (i === 0) {
            result.push([1]);
        } else if (i === 1) {
            result.push([1, 1]);
        } else {
            result.push([]);
            for (let j = 0; j <= i; j++) {
                if (j === 0 || j === i) {
                    result[i].push(1);
                } else {
                    result[i].push(result[i - 1][j - 1] + result[i - 1][j]);
                }
            }
        }
    }
    console.log(result);
    return result;
};
```
