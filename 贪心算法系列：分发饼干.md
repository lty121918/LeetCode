# 贪心算法系列：分发饼干

```js
/**
 * @param {number[]} g
 * @param {number[]} s
 * @return {number}
 */
var findContentChildren = function (g, s) {
   let i = 0;
   let res = 0;
   g.sort((a, b) => a - b);
   s.sort((a, b) => a - b);
   for (let j = 0; j < s.length; j++) {
      if (g[i] <= s[j]) {
         i++;
         res++;
      }

   }
   return res;
};
```
